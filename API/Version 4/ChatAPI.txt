public function broadcast($message)
	Description:
		Broadcasts a message to every player on the server
	Parameters:
		$message = Message to send
	Return Values:
		None

public function sendTo($owner, $text, $player)
	Description:
		Sends a message to a specific player
	Parameters:
		$owner = Owner of the message
		$text = The Actual text to send
		$player = The Player to send the message to
	Return Values:
		None

public function send($owner, $text, $whitelist = false, $blacklist = false)
	Description:
		Sends a message to every player on the server but specifying who sent the message
	parameters:
		$owner = The person who (supposedly) sent the message
		$text = The message itself
		$whitelist = Unknown
		$blacklist = Unknown
