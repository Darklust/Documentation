public function isOp($username)
	Description:
		Checks if a user is an OP
	Parameters:
		$username = username to check
	Return Values:
		TRUE if OP
		FALSE if else

public function ban($username)
	Description:
		Bans a username
	Parameters:
		$username = username to ban
	Return Values:
		None

public function pardon($username)
	Description:
		Unbans a username
	Parameters:
		$username = username to ban
	Return Values:
		None
		
public function banIP($ip)
	Description:
		IPbans an IP
	Parameters:
		$ip = ip to ban
	Return Values:
		None
		
public function pardonIP($ip)
	Description:
		IPpardons an IP
	Parameters:
		$ip = ip to pardon
	Return Values:
		None
		
public function kick($username, $reason)
	Description:
		Kicks a person out of the server
	Parameters:
		$username = username of a player to kick
		$reason = reason for kicking the player
	Return Values:
		None

public function reload()
	Description:
		Reloads the ban configuration files
	Parameters:
		None
	Return Values:
		None
		
public function isIPBanned($ip)
	Description:
		Checks if a player is IPBanned
	Parameters:
		$ip = ip of a player to check for ban
	Return Values
		true = if banned
		false = if not banned

public function isBanned($username)
	Description:
		Checks if a player Banned
	Parameters:
		$username = username of a player to check for ban
	Return Values
		true = Banned
		false = Not banned
		
public function inWhitelist($username)
	Description:
		Checks if a player Whitelisted
	Parameters:
		$username = username of a player to check for whitelist
	Return Values
		true = Whitelisted
		false = Not Whitelisted
